Проблема: при экспорте не находит исходный файл.

Исправь путь к файлу — проверять несколько вариантов:

import os
from pathlib import Path

def find_result_file(filename):
    appdata = os.environ.get('APPDATA', '')
    candidates = [
        os.path.join(appdata, 'Transkrib', 'storage', 'results', filename),
        os.path.join(os.path.dirname(__file__), '..', '..', 'storage', 'results', filename),
        os.path.join('storage', 'results', filename),
    ]
    for path in candidates:
        normalized = os.path.normpath(path)
        if os.path.exists(normalized):
            return normalized
    return None

Используй find_result_file() перед экспортом.
